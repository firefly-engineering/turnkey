# Design: use_turnkey direnv function

## Overview

Replace inline .envrc logic with a generated `use_turnkey` function that:
1. Auto-regenerates dependency files based on declared toolchains
2. Syncs Buck2 cell symlinks
3. Declares appropriate watch_file entries

## Target .envrc

```bash
use flake . --no-pure-eval
source "$TURNKEY_DIRENV_LIB"
use turnkey
```

## API Design

### Environment Variables (set by flake)

| Variable | Description |
|----------|-------------|
| `TURNKEY_DIRENV_LIB` | Path to generated shell library |
| `TURNKEY_ENABLED_LANGS` | Space-separated list: "go python rust" |
| `TURNKEY_GODEPS_GEN` | Path to godeps-gen binary |
| `TURNKEY_PYDEPS_GEN` | Path to pydeps-gen binary |
| `TURNKEY_RUSTDEPS_GEN` | Path to rustdeps-gen binary |
| `TURNKEY_UV` | Path to uv binary |
| `TURNKEY_BUCK2_CONFIG` | Path to generated .buckconfig |
| `TURNKEY_BUCK2_TOOLCHAINS_CELL` | Path to toolchains cell |
| `TURNKEY_CELL_<NAME>` | Path:derivation for each deps cell |

### Generated Script Structure

```bash
# turnkey-direnv-lib.sh (generated by Nix)

TURNKEY_DEPS_CACHE=".turnkey-deps-hashes"

# === Helper Functions (always included) ===

_turnkey_hash_files() { ... }
_turnkey_get_cached_hash() { ... }
_turnkey_set_cached_hash() { ... }
_turnkey_needs_regen() { ... }

# === Language-specific regeneration (generated per toolchain) ===

_turnkey_regen_go() {
  if [ ! -f go.mod ]; then return; fi
  if _turnkey_needs_regen "go" "go-deps.toml" go.mod go.sum; then
    echo "turnkey: Regenerating go-deps.toml..."
    if "$TURNKEY_GODEPS_GEN" --go-mod go.mod --go-sum go.sum --prefetch -o go-deps.toml.tmp; then
      mv go-deps.toml.tmp go-deps.toml
      _turnkey_set_cached_hash "go" "$(_turnkey_hash_files go.mod go.sum)"
      echo "turnkey: Updated go-deps.toml (remember to commit)"
    else
      rm -f go-deps.toml.tmp
      echo "turnkey: Warning: Failed to generate go-deps.toml"
    fi
  fi
}

_turnkey_regen_python() {
  if [ ! -f pyproject.toml ]; then return; fi
  if _turnkey_needs_regen "python" "python-deps.toml" pyproject.toml; then
    echo "turnkey: Regenerating python-deps.toml..."
    if "$TURNKEY_UV" lock && \
       "$TURNKEY_UV" export --format pylock.toml -o pylock.toml && \
       "$TURNKEY_PYDEPS_GEN" --lock pylock.toml -o python-deps.toml; then
      _turnkey_set_cached_hash "python" "$(_turnkey_hash_files pyproject.toml)"
      echo "turnkey: Updated python-deps.toml"
    else
      echo "turnkey: Warning: Failed to generate python-deps.toml"
    fi
  fi
}

_turnkey_regen_rust() {
  if [ ! -f Cargo.lock ]; then return; fi
  if _turnkey_needs_regen "rust" "rust-deps.toml" Cargo.toml Cargo.lock; then
    echo "turnkey: Regenerating rust-deps.toml..."
    if "$TURNKEY_RUSTDEPS_GEN" --cargo-lock Cargo.lock -o rust-deps.toml; then
      _turnkey_set_cached_hash "rust" "$(_turnkey_hash_files Cargo.toml Cargo.lock)"
      echo "turnkey: Updated rust-deps.toml"
    else
      echo "turnkey: Warning: Failed to generate rust-deps.toml"
    fi
  fi
}

# === Symlink syncing ===

_turnkey_sync_symlinks() {
  # .buckconfig
  if [ -n "$TURNKEY_BUCK2_CONFIG" ]; then
    if [ "$(readlink .buckconfig 2>/dev/null)" != "$TURNKEY_BUCK2_CONFIG" ]; then
      ln -sf "$TURNKEY_BUCK2_CONFIG" .buckconfig
      echo "turnkey: Synced .buckconfig"
    fi
  fi

  # Toolchains cell
  if [ -n "$TURNKEY_BUCK2_TOOLCHAINS_CELL" ]; then
    mkdir -p .turnkey
    if [ "$(readlink .turnkey/toolchains 2>/dev/null)" != "$TURNKEY_BUCK2_TOOLCHAINS_CELL" ]; then
      ln -sfn "$TURNKEY_BUCK2_TOOLCHAINS_CELL" .turnkey/toolchains
      echo "turnkey: Synced toolchains cell"
    fi
  fi

  # Dynamic cells (TURNKEY_CELL_*)
  for var in $(env | grep '^TURNKEY_CELL_' | cut -d= -f1); do
    local value="${!var}"
    local cell_path="${value%%:*}"
    local cell_deriv="${value#*:}"
    if [ "$(readlink "$cell_path" 2>/dev/null)" != "$cell_deriv" ]; then
      mkdir -p "$(dirname "$cell_path")"
      ln -sfn "$cell_deriv" "$cell_path"
      echo "turnkey: Synced $cell_path"
    fi
  done
}

# === Main entry point ===

use_turnkey() {
  local skip_regen=0 skip_sync=0 skip_watch=0
  local only_langs="" skip_langs=""

  # Parse flags
  while [[ $# -gt 0 ]]; do
    case "$1" in
      --skip-regen) skip_regen=1 ;;
      --skip-sync)  skip_sync=1 ;;
      --skip-watch) skip_watch=1 ;;
      --only-go)    only_langs="$only_langs go" ;;
      --only-python) only_langs="$only_langs python" ;;
      --only-rust)  only_langs="$only_langs rust" ;;
      --skip-go)    skip_langs="$skip_langs go" ;;
      --skip-python) skip_langs="$skip_langs python" ;;
      --skip-rust)  skip_langs="$skip_langs rust" ;;
      *) echo "turnkey: Unknown flag: $1" >&2 ;;
    esac
    shift
  done

  # Check env vars (flags take precedence)
  [[ -n "$TURNKEY_SKIP_REGEN" ]] && skip_regen=1

  # Build list of languages to process
  local langs_to_process=""

  if [[ -n "$only_langs" ]]; then
    # Flags: --only-go etc take highest precedence
    langs_to_process="$only_langs"
  elif [[ -n "$TURNKEY_SKIP_ALL" ]]; then
    # SKIP_ALL mode: start empty, add ENABLE_* languages
    [[ -n "$TURNKEY_ENABLE_GO" ]] && langs_to_process="$langs_to_process go"
    [[ -n "$TURNKEY_ENABLE_PYTHON" ]] && langs_to_process="$langs_to_process python"
    [[ -n "$TURNKEY_ENABLE_RUST" ]] && langs_to_process="$langs_to_process rust"
  else
    # Default mode: start with all, remove SKIP_* languages
    langs_to_process="$TURNKEY_ENABLED_LANGS"
    [[ -n "$TURNKEY_SKIP_GO" ]] && skip_langs="$skip_langs go"
    [[ -n "$TURNKEY_SKIP_PYTHON" ]] && skip_langs="$skip_langs python"
    [[ -n "$TURNKEY_SKIP_RUST" ]] && skip_langs="$skip_langs rust"
  fi

  # Regenerate deps
  if [[ "$skip_regen" != "1" ]]; then
    for lang in $langs_to_process; do
      if [[ ! " $skip_langs " =~ " $lang " ]]; then
        "_turnkey_regen_$lang"
      fi
    done
  fi

  # Sync symlinks
  if [[ "$skip_sync" != "1" ]]; then
    _turnkey_sync_symlinks
  fi

  # Watch files
  if [[ "$skip_watch" != "1" ]]; then
    watch_file nix/**/*.nix
    watch_file nix/**/*.patch
    # Language-specific watch files (generated per enabled language)
  fi
}
```

## Nix Generation

The script is generated in `nix/flake-parts/turnkey/default.nix`:

```nix
# Pseudo-code for script generation
turnkeyDirenvLib = pkgs.writeShellScript "turnkey-direnv-lib.sh" ''
  ${helpers}  # Always included

  ${lib.optionalString (hasGo config) goRegenFunction}
  ${lib.optionalString (hasPython config) pythonRegenFunction}
  ${lib.optionalString (hasRust config) rustRegenFunction}

  use_turnkey() {
    ${lib.optionalString (hasGo config) "_turnkey_regen_go"}
    ${lib.optionalString (hasPython config) "_turnkey_regen_python"}
    ${lib.optionalString (hasRust config) "_turnkey_regen_rust"}
    _turnkey_sync_symlinks

    watch_file nix/**/*.nix
    watch_file nix/**/*.patch
    ${lib.optionalString (hasGo config) ''
      watch_file go.mod
      watch_file go.sum
    ''}
    ${lib.optionalString (hasPython config) ''
      watch_file pyproject.toml
    ''}
    ${lib.optionalString (hasRust config) ''
      watch_file Cargo.toml
      watch_file Cargo.lock
    ''}
  }
'';
```

## Language Detection

A toolchain enables a language if:
1. It appears in toolchain.toml under `[toolchains]`
2. It's one of: `go`, `python`, `rust`

The `*deps-gen` tools are separate toolchains but their presence isn't required - the language toolchain is what matters.

## Extensibility

Adding a new language requires:
1. Create `<lang>deps-gen` tool
2. Add language to the generator in `default.nix`
3. Define: input files, output file, regeneration command

## Migration Path

1. Implement generation in Nix
2. Add `TURNKEY_DIRENV_LIB` to devenv environment
3. Update template .envrc to use new pattern
4. Update turnkey's own .envrc as proof of concept

## CLI Flags

`use_turnkey` accepts optional flags:

```bash
use turnkey                     # Default: all features enabled
use turnkey --skip-regen        # Skip dependency regeneration
use turnkey --skip-sync         # Skip symlink syncing
use turnkey --skip-watch        # Skip watch_file declarations
use turnkey --only-go           # Only process Go (can combine: --only-go --only-rust)
use turnkey --skip-go           # Skip Go, process others
```

## Environment Variables for CI

Per-language control via environment variables (useful for CI partial builds):

```bash
# Skip all languages, then selectively enable (recommended for CI)
TURNKEY_SKIP_ALL=1           # Skip all language regeneration
TURNKEY_ENABLE_GO=1          # Enable Go (only meaningful with SKIP_ALL)
TURNKEY_ENABLE_PYTHON=1      # Enable Python
TURNKEY_ENABLE_RUST=1        # Enable Rust

# Or skip specific languages (default is all enabled)
TURNKEY_SKIP_GO=1            # Skip Go regeneration
TURNKEY_SKIP_PYTHON=1        # Skip Python regeneration
TURNKEY_SKIP_RUST=1          # Skip Rust regeneration

# Skip all regeneration entirely (equivalent to --skip-regen)
TURNKEY_SKIP_REGEN=1

# Example CI usage: only check Go portion of monorepo
TURNKEY_SKIP_ALL=1 TURNKEY_ENABLE_GO=1 nix develop --command tk build //go/...

# Example: check everything except Python (slow deps)
TURNKEY_SKIP_PYTHON=1 nix develop --command tk test //...
```

**Precedence:**
1. Flags (highest)
2. `TURNKEY_ENABLE_*` (explicit enable overrides skip)
3. `TURNKEY_SKIP_*` / `TURNKEY_SKIP_ALL`
4. Default (all enabled)

## Decisions

1. **Custom regeneration hooks**: Deferred. Need more experience with standard workflows first before designing extension points.

2. **Flag vs env var precedence**: Flags override env vars. Env vars provide defaults.

3. **Partial monorepo CI**: Supported via `TURNKEY_SKIP_<LANG>` env vars.
